```python
#

# 
# @param arr int整型一维数组 the array
# @return int整型
#
class Solution:
    def maxLength(self , arr: List[int]) -> int:
        # write code here
        map_ = {}
        start = 0
        maxLen = 1
        map_[arr[0]] = 0
        for i in range(1, len(arr)):
#             print('arr[i]', arr[i])
#             print(map_)
            if arr[i] in map_:
                start = max(map_[arr[i]]+1,start) #这里注意
            map_[arr[i]] = i
            maxLen = max(maxLen, i-start+1)
#             print('start', start)
#             print('maxLen', maxLen)
            
        return maxLen
```
