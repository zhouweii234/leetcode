class Solution:
    def sortedSquares(self, nums: List[int]) -> List[int]:
        left = 0
        right = len(nums)-1
        cur = len(nums)-1
        ans = [0]*len(nums)

        while left <= right:
            leftNum = nums[left]**2
            rightNum = nums[right]**2

            if leftNum <= rightNum:
                ans[cur] = rightNum
                right -= 1
            else:
                ans[cur] = leftNum
                left += 1
            cur -= 1
        
        return ans
