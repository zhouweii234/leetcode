349_两个数组的交集
===
···python
class Solution(object):
    def intersection(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: List[int]
        """
        res = list(set(nums1)&set(nums2))
        return res
