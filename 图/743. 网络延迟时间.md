class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        graph = [[float("INF")]*(n+1) for _ in range(n+1)]
        for u, v, i in times:
            graph[u][v] = i
        graph[k][k] = 0

        minNode = k
        mindis = float("INF")
        unmarked = [i for i in range(1, n+1)]

        for _ in range(1,n):
            if minNode in unmarked:
                unmarked.remove(minNode)
            for i in range(1,n+1):
                if i == k:
                    continue
                graph[k][i] = min(graph[k][i], graph[k][minNode]+graph[minNode][i])
            
            mindis = float("INF")
            for i in unmarked:
                if graph[k][i] < mindis:
                    mindis = graph[k][i]
                    minNode = i

        maxdis = 0
        maxNode = k
        for i in range(1,n+1):
            if i == k:
                continue
            # print('i', i)
            # print('graph[k][i]', graph[k][i])
            if graph[k][i] > maxdis:
                maxdis = graph[k][i]
                maxNode = i
        
        if maxdis == float("INF"):
            return -1
        else:
            return maxdis
